📝 HTML Implementation

Step 1: Include CSS and JavaScript
<!-- Include CSS -->
<link rel="stylesheet" href="assets/css/car-series-selector.css">

<!-- Include JavaScript -->
<script src="assets/js/car-series-selector.js"></script>
Step 2: Create HTML Container
<!-- Basic Usage -->
<div data-car-series-selector="honda-civic"></div>

<!-- With Default Series -->
<div data-car-series-selector="toyota-corolla" data-default-series="corolla-gli"></div>

<!-- With Custom ID -->
<div id="car-series" data-car-series-selector="honda-civic"></div>
Step 3: Initialize (Optional)
<!-- Auto-initializes on DOM ready -->
<script>
// Manual initialization (optional)
window.AutoWorldCarSeries.init();
</script>

⚡ JavaScript API

Get Selected Series
const selectedSeries = window.AutoWorldCarSeries.getSelectedSeries('honda-civic');
console.log(selectedSeries); // "civic-vti"
Get Series Data
const seriesData = window.AutoWorldCarSeries.getCarSeriesData('honda-civic');
console.log(seriesData); // Full car data object
Set Series Programmatically
window.AutoWorldCarSeries.setSelectedSeries('honda-civic', 'civic-rs');
Listen for Series Changes
const container = document.getElementById('car-series');
container.addEventListener('seriesChanged', function(event) {
  const { carId, seriesId, seriesData } = event.detail;
  console.log('Series changed to:', seriesData.name);
});
